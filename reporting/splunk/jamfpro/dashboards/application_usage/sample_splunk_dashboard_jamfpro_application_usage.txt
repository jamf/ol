| join computer.pagination.event_id 
    [ search computer.location.username=* ] 
| rename computer.location.username as Username
| join type=left id [search "source"="jamf_app_usage" | stats sum(usage{}.apps{}.foreground) as Minutes by usage{}.apps{}.name, id | rename usage{}.apps{}.name as App | where App="Microsoft Word" | fields id, Minutes ] 
| table Username, Serial_Number, Minutes

| sort Minutes</query>
